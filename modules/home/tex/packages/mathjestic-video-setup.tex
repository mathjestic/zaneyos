% --- cancel helpers ---
\ProvideDocumentCommand{\rcancel}{ O{3mu} m m }{%
  \renewcommand\CancelColor{\color{red}}%
  \cancel{#2}^{\color{red}{\mkern#1 #3}}%
}
\ProvideDocumentCommand{\lrcancel}{ O{3mu} m m }{%
  % using 'green' so this works even if dvipsnames wasn't applied
  \renewcommand\CancelColor{\color{green}}%
  \bcancel{#2}^{\color{green}{\mkern#1 #3}}%
}
